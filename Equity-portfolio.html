Calculating the dividend for all stocks

In [21]:
 
stock_names=c("IBM","AAPL","AMZN","FB","GOOG","MSFT","NFLX","ORCL","SAP","TSLA")
for (name in 1:length(stock_names)){
    row_indx=c()
    dividend=c()
    close_price=stock[[paste(stock_names[name],"_","Close",sep="")]] # take very first row value of close
    adj_close_price=stock[[paste(stock_names[name],"_","Adj.Close",sep="")]] # take very first row value of close
    for (row in 1:(nrow(stock)-1)){
        if (round(close_price[row]/close_price[row+1],digits=3)!=round(adj_close_price[row]/adj_close_price[row+1],digits=3)){
            row_indx=append(row_indx,row+1)
            dividend_total=round(abs((close_price[row]/close_price[row+1])-(adj_close_price[row]/adj_close_price[row+1]))*close_price[row+1],digits=4)
            dividend=append(dividend,dividend_total)
        }
    }
    
    if(length(row_indx)!=0){
        dividend_col=data.frame("dividend"=replace(c(rep(0,nrow(stock))),row_indx,dividend))        
    }else{
        dividend_col=data.frame("dividend"=c(rep(0,nrow(stock))))  
    }
    colnames(dividend_col)=paste(stock_names[name],"_","dividend",sep="")
    stock=cbind(stock,dividend_col)
}

# Stragety = FALSE  --> rebalancing HIGH
# Stragety = TRUE  --> rebalancing LOW
# Balancing_day = numeric input ----> Days interval for rebalancing
rebalancing=function(strategy, balancing_day){
    rebalancing_day=balancing_day
    MTM=data.frame()
    df_dropped_price=data.frame()
    df_adj_close = select(stock,ends_with("Adj.Close"))
    df_close = select(stock,ends_with("Close"))
    new_portfolio=portfolio
    first_MTM=5000000
    MTM=rbind(MTM,first_MTM)
    day= seq(1, 251, by=rebalancing_day)
    for(row in 1:(length(day)-1)){
        cash_total=sum(new_portfolio["cash"]) 
        # SELLING STOCK On REBALNCING DAY
        first_stocks=new_portfolio[,"stocks"]
        for(i in 1:nrow(new_portfolio)){
            stock_day=df_close[day[row+1],]
            close_price=stock_day[[paste(first_stocks[i],"_","Close",sep="")]] # take very first row value of close
            #cash account with dividends
            cash_account=stock[,paste(first_stocks[i],"_","dividend",sep="")][day[row+1]]*new_portfolio[new_portfolio$stocks==first_stocks[i],][["shares"]]
            cash_total=cash_total+new_portfolio[new_portfolio$stocks==first_stocks[i],][["shares"]]*close_price+cash_account
        }   
        MTM=rbind(MTM,cash_total) 
        ##### Buy shares on the five stocks with the most price dropped 
        dropped_price=(df_adj_close[day[row+1],]-df_adj_close[day[row],])*100/(df_adj_close[day[row],])
        df_dropped_price=rbind(df_dropped_price,dropped_price)
        day_percent=sort(df_dropped_price[row,], decreasing = strategy)
        stocks=c()
        for(string in 1:5){
            stock_name=unlist(strsplit(colnames(day_percent[string]),"_"))[1]
            stocks=append(stocks,stock_name)
        }   
        new_portfolio["stocks"]=stocks
        cash=c()
        shares=c()
        for(i in 1:length(stocks)){
            stock_day=stock[day[row+1],]
            close_price=stock_day[[paste(stocks[i],"_","Close",sep="")]][1] # take very first row value of close
            cash_per_stock=cash_total/5
            shares=append(shares,floor(cash_per_stock/close_price))
            cash=append(cash,cash_per_stock-shares[i]*close_price)
        }
        new_portfolio["shares"]=shares
        new_portfolio["cash"]=cash
    }
    MTM=cbind(MTM,stock$Date[day])
    colnames(MTM)=c("MTM_series","Date")
    MTM
}
